Add to the existing `window.BX` namespace (do NOT overwrite it — extend it):

```javascript
// Commission calculation — pure display functions
// Actual commission is calculated server-side in RPCs (per D-04)
(function(BX) {
  BX.calcMatchCommission = function(netPnl, rate) {
    // Match commission: rate% of net loss (zero if winner)
    if (netPnl >= 0) return 0;
    return Math.abs(netPnl) * (rate / 100);
  };

  BX.calcFancyCommission = function(volume, rate) {
    // Fancy commission: rate% of total volume (regardless of win/loss)
    return Math.abs(volume) * (rate / 100);
  };
})(window.BX = window.BX || {});
```

These are reference/preview functions only. The real commission is in PostgreSQL RPCs.
